Dynamic Variables:

Variables can change not only in value but also in type. For instance, a variable initially declared as a string can later be assigned an integer value.

var age = 25;
// Alternatively, you can use the "let" keyword:
let someVariable = 25;

age = "dynamic typed languages are intriguing";
console.log(age); // outputs "dynamic typed languages are intriguing"

Constant Variables:

const salary = 8000;
console.log(salary);

Strings:

You can use double quotes (""), single quotes (''), or backticks (``) for strings. The advantage of backticks is template string functionality, which allows for variable interpolation with ${}.

var name = "John";
var language = "Javascript";
var anotherString = `what? ${name}`;
console.log(name + language + anotherString); // JohnJavascriptwhat? John
console.log(name, language, anotherString);   // John Javascript what? John

Undefined Value:

var result; // This is "undefined". You can also explicitly set it: var result = undefined;

Null Value:

Unlike undefined, you have to explicitly set a variable to null.

var result2 = null;

Objects:

Here's how to declare an object with properties.

var person = {
  first_name: "Bruce",
  last_name: "Wayne",
  age: 30,
};
console.log(person.first_name);

Arrays:

const oddNumbers = [1, 3, 5, 7, 9];
console.log(oddNumbers[2]);  // Outputs 5

Comparison Operators:

== checks if values are equivalent, while === checks for both value and type equivalence.

var x = 5;
var z = "5";
console.log(x == z);  // true
console.log(x === z); // false

Ternary Operators:

It follows the pattern: condition ? value_if_true : value_if_false.

var isEven = 10 % 2 === 0 ? "yes" : "no";
console.log(isEven); // yes

Type Conversion:

console.log(parseFloat("3.14"));
console.log(parseInt("3"));
console.log(String(543.123));
console.log((543.123).toString());

Functions:

Traditional function:

function sum(a, b) {
    return a + b;
}
var summation = sum(5, 4);
console.log(summation);

Arrow function:

var arrowSum = (a, b) => {
    return a + b;
}
// Shorter syntax:
// var arrowSum = (a, b) => a + b;
var arrowSumation = arrowSum(5, 4);
console.log(arrowSumation);

Array Methods:

forEach(): This method allows iterating over each element in the array.

var numbers = [1, 2, 3, 4, 5];
numbers.forEach((element, index, array) => {
  console.log(`${element} is at index ${index} in the array ${array}`);
});

map(): Returns a new array with a function applied to each of its elements.

var numbers = [1, 2, 3, 4, 5];
var squares = numbers.map(element => element * element);
console.log(squares); // [1, 4, 9, 16, 25]

filter(): Returns a new array containing all elements for which the provided function returns a truthy value.

var numbers = [1, 2, 3, 4, 5];
var evenNumbers = numbers.filter(element => element % 2 === 0);
console.log(evenNumbers); // [2, 4]

reduce(): Applies a function against an accumulator and each element in the array to reduce it to a single value.

var numbers = [1, 2, 3, 4, 5];
var sum = numbers.reduce((accumulator, currentValue) => accumulator + currentValue, 0);
console.log(sum); // 15

find(): Returns the first element in the array that satisfies the provided testing function.

var numbers = [1, 2, 3, 4, 5];
var firstEvenNumber = numbers.find(element => element % 2 === 0);
console.log(firstEvenNumber); // 2

indexOf(): Returns the first index at which a certain element can be found in the array.

var colors = ["red", "green", "blue", "yellow"];
var indexOfBlue = colors.indexOf("blue");
console.log(indexOfBlue); // 2

slice(): Returns a shallow copy of a portion of an array into a new array.

var fruits = ["apple", "banana", "cherry", "date"];
var selectedFruits = fruits.slice(1, 3);
console.log(selectedFruits); // ["banana", "cherry"]

splice(): Changes the contents of an array by removing, replacing, or adding elements.

var fruits = ["apple", "banana", "cherry", "date"];
fruits.splice(2, 1, "mango");
console.log(fruits); // ["apple", "banana", "mango", "date"]

Promises and Asynchronous Programming:

Asynchronous programming allows for non-blocking code execution. Promises offer a way to handle the results of asynchronous operations.

Example of asynchronous function:

function getDataFromServer() {
    setTimeout(() => {
        console.log("Data received from server");
    }, 2000);
}
console.log("Starting to get data...");
getDataFromServer();
console.log("Data processing completed.");

Example of a Promise:

function sendDataToServer(data) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            if (data) {
                resolve("Data sent successfully");
            } else {
                reject("Error sending data");
            }
        }, 1000);
    });
}

let userInput = "example data";
sendDataToServer(userInput)
    .then(response => console.log(response))
    .catch(error => console.log(error));

Fetch API:

Allows for making network requests from the client side. Here are CRUD examples using the Fetch API:

// GET request
fetch("https://example.com/users")
    .then((response) => response.json())
    .then((data) => {
        console.log("GET request data:", data);
    })
    .catch((error) => {
        console.log("GET request error:", error);
    });

// POST request
fetch("https://example.com/users", {
    method: "POST",
    body: JSON.stringify({ name: "John", age: 22, email: "john@example.com" }),
    headers: { "Content-Type": "application/json" },
})
    .then((response) => response.json())
    .then((data) => {
        console.log("POST request data:", data);
    })
    .catch((error) => {
        console.log("POST request error:", error);
    });

// PUT request
fetch("https://example.com/users/1", {
    method: "PUT",
    body: JSON.stringify({ name: "Jane", age: 25 }),
    headers: { "Content-Type": "application/json" },
})
    .then((response) => response.json())
    .then((data) => {
        console.log("PUT request data:", data);
    })
    .catch((error) => {
        console.log("PUT request error:", error);
    });

// DELETE request
fetch("https://example.com/users/1", {
    method: "DELETE",
})
    .then((response) => response.json())
    .then((data) => {
        console.log("DELETE request data:", data);
    })
    .catch((error) => {
        console.log("DELETE request error:", error);
    });
